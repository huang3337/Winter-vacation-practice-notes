1. 首先打开CubeMX，点击HELP、点击Connection & Updates

13. 添加一个headfile.h用于集合所有的头文件，其他的.c文件只用#include "headfile.h"就好了

    ![](./%E6%96%B0%E5%BB%BA%E6%96%87%E4%BB%B6%E7%9A%84%E6%93%8D%E4%BD%9C.assets/ScreenShot_2026-03-06_141711_510.png)

#ifndef _headfile_h
#define _headfile_h

#include "stm32g4xx.h"                  // Device header
#include "stdio.h"
#include "string.h"
#include "stdint.h"

#include "main.h"
#include "gpio.h"

/*
这里是自己要加入的其他头文件
#include "fun.h"
#include "lcd.h"
*/

#endif
